local function getTargetPart(player)
    local character = player.Character
    if not character then return nil end
    local part = character:FindFirstChild(SilentAimSettings.HitPart)
    if not part then
        part = character:FindFirstChild("Head")
    end
    return part
end

local function isAlive(player)
    local character = player.Character
    if not character then return false end
    local humanoid = character:FindFirstChildWhichIsA("Humanoid")
    if not humanoid then return false end
    return humanoid.Health > 0
end

local function isVisible(origin, targetPart, originalIgnore)
    if not SilentAimSettings.WallCheck then
        return true
    end
    local direction = targetPart.Position - origin
    local distance = direction.Magnitude
    if distance <= 0 then return false end

    local raycastParams = RaycastParams.new()
    raycastParams.FilterType = Enum.RaycastFilterType.Blacklist
    local ignoreList = { LocalPlayer.Character }

    local targetChar = targetPart.Parent
    if targetChar then
        table.insert(ignoreList, targetChar)
    end

    if originalIgnore then
        if type(originalIgnore) == "table" then
            for _, v in ipairs(originalIgnore) do table.insert(ignoreList, v) end
        else
            table.insert(ignoreList, originalIgnore)
        end
    end

    raycastParams.FilterDescendantsInstances = ignoreList

    local result = Workspace:Raycast(origin, direction, raycastParams)
    if not result then
        return true
    end

    local hitInstance = result.Instance
    if hitInstance == targetPart or (targetChar and hitInstance:IsDescendantOf(targetChar)) then
        return true
    end

    return false
end

local function getClosestPlayerToMouse()
    local shortestDist = math.huge
    local closest = nil
    local mousePos = UserInputService:GetMouseLocation()

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and isAlive(player) then
            if player.Team == LocalPlayer.Team and player.Team ~= nil and not SilentAimSettings.ShowTeammates then
                continue
            end

            local hitPart = getTargetPart(player)
            if hitPart then
                local screenPos, onScreen = Camera:WorldToViewportPoint(hitPart.Position)
                if onScreen then
                    local dist = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
                    
                    if SilentAimSettings.FOVEnabled then
                        if dist <= SilentAimSettings.FOVRadius and dist < shortestDist then
                            shortestDist = dist
                            closest = player
                        end
                    else
                        if dist < shortestDist then
                            shortestDist = dist
                            closest = player
                        end
                    end
                end
            end
        end
    end
    return closest
end

-- Intercept the game's Raycast functionality for Silent Aim
Utility.Raycast = function(self, from, to, range, ignore, mode, debug)
    if not SilentAimSettings.Enabled then
        return OriginalRaycast(self, from, to, range, ignore, mode, debug)
    end

    if math.random(1, 100) > SilentAimSettings.HitChance then
        return OriginalRaycast(self, from, to, range, ignore, mode, debug)
    end

    local target = getClosestPlayerToMouse()
    if not target then
        return OriginalRaycast(self, from, to, range, ignore, mode, debug)
    end

    local hitPart = getTargetPart(target)
    if not hitPart then
        return OriginalRaycast(self, from, to, range, ignore, mode, debug)
    end

    if not isVisible(from, hitPart, ignore) then
        return OriginalRaycast(self, from, to, range, ignore, mode, debug)
    end

    local hitPosition = hitPart.Position
    local direction = (hitPosition - from).Unit
    local distance = (hitPosition - from).Magnitude

    if distance > range then
        distance = range
        hitPosition = from + direction * range
    end

    return {
        Position = hitPosition,
        Distance = distance,
        Instance = hitPart,
        Material = hitPart.Material,
        Normal = -direction,
    }
end

local function toggleTableAttribute(attribute, value)
    for _, gcVal in pairs(getgc(true)) do
        if type(gcVal) == "table" and rawget(gcVal, attribute) then
            gcVal[attribute] = value
        end
    end
end

local ESP_Cache = {}

local R15_Bones = {
    Spine = {{"UpperTorso", "LowerTorso"}, {"UpperTorso", "Head"}},
    Limbs = {
        {"UpperTorso", "LeftUpperArm"}, {"LeftUpperArm", "LeftLowerArm"}, {"LeftLowerArm", "LeftHand"},
        {"UpperTorso", "RightUpperArm"}, {"RightUpperArm", "RightLowerArm"}, {"RightLowerArm", "RightHand"},
        {"LowerTorso", "LeftUpperLeg"}, {"LeftUpperLeg", "LeftLowerLeg"}, {"LeftLowerLeg", "LeftFoot"},
        {"LowerTorso", "RightUpperLeg"}, {"RightUpperLeg", "RightLowerLeg"}, {"RightLowerLeg", "RightFoot"}
    }
}
local R6_Bones = {
    Spine = {{"Torso", "Head"}},
    Limbs = {
        {"Torso", "Left Arm"}, {"Torso", "Right Arm"},
        {"Torso", "Left Leg"}, {"Torso", "Right Leg"}
    }
}

local function CachePlayerParts(data, char, rigType)
    data.CachedParts = {}
    if not char then return end
    
    local structure = (rigType == Enum.HumanoidRigType.R15) and R15_Bones or R6_Bones
    data.CachedParts["HumanoidRootPart"] = char:FindFirstChild("HumanoidRootPart")

    local function checkAndCache(name)
        if not data.CachedParts[name] then
            data.CachedParts[name] = char:FindFirstChild(name)
        end
    end

    for _, pair in ipairs(structure.Spine) do checkAndCache(pair[1]) checkAndCache(pair[2]) end
    for _, pair in ipairs(structure.Limbs) do checkAndCache(pair[1]) checkAndCache(pair[2]) end
end

local function RemovePlayerESP(player)
    local data = ESP_Cache[player]
    if data then
        data.BoxOutline:Destroy() data.Box:Destroy()
        data.HealthBarBg:Destroy() data.HealthBar:Destroy()
        data.NameText:Destroy() data.WeaponText:Destroy() data.DistanceText:Destroy()
        for i = 1, #data.SkeletonLines do data.SkeletonLines[i]:Destroy() end
        if data.Highlight then data.Highlight:Destroy() end
        ESP_Cache[player] = nil
    end
end

local function AddPlayerESP(player)
    if player == LocalPlayer then return end
    
    local data = {
        BoxOutline = Drawing.new("Square"), Box = Drawing.new("Square"),
        HealthBarBg = Drawing.new("Line"), HealthBar = Drawing.new("Line"),
        NameText = Drawing.new("Text"), WeaponText = Drawing.new("Text"), DistanceText = Drawing.new("Text"),
        SkeletonLines = {}, CachedParts = {}, RigType = Enum.HumanoidRigType.R6
    }

    data.BoxOutline.Filled = false data.BoxOutline.Thickness = 3
    data.Box.Filled = false data.Box.Thickness = 1
    data.HealthBarBg.Thickness = 3 data.HealthBarBg.Color = Color3.fromRGB(0, 0, 0)
    data.HealthBar.Thickness = 1
    data.NameText.Center = true data.NameText.Outline = true
    data.WeaponText.Center = true data.WeaponText.Outline = true
    data.DistanceText.Center = true data.DistanceText.Outline = true

    for i = 1, 14 do
        local line = Drawing.new("Line")
        line.Visible = false
        data.SkeletonLines[i] = line
    end

    data.Highlight = Instance.new("Highlight")
    data.Highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    data.Highlight.Parent = CoreGui

    local function setupChar(char)
        if not char then return end
        local hum = char:WaitForChild("Humanoid", 5)
        if hum then
            data.RigType = hum.RigType
            CachePlayerParts(data, char, hum.RigType)
        end
    end

    player.CharacterAdded:Connect(setupChar)
    if player.Character then setupChar(player.Character) end

    ESP_Cache[player] = data
end

-- MASTER RUNTIME RUNSTEPMOUNT LOOPS
-- =============================================================================
RunService.RenderStepped:Connect(function()
    local mouseLoc = UserInputService:GetMouseLocation()
    local isRightClickHolding = UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2)

    -- 1. FOV Ring Refresh Processing
    if SilentAimSettings.Enabled and SilentAimSettings.FOVEnabled then
        FOVCircle.Position = mouseLoc
        FOVCircle.Radius = SilentAimSettings.FOVRadius
        FOVCircle.Color = SilentAimSettings.FOVColor
        FOVCircle.Thickness = SilentAimSettings.FOVThickness
        FOVCircle.Filled = SilentAimSettings.FOVFilled
        FOVCircle.Transparency = SilentAimSettings.FOVTransparency
        FOVCircle.Visible = true
    else
        FOVCircle.Visible = false
    end

    -- 2. Integrated Aimbot & Triggerbot Core Logic Processing from file
    if isRightClickHolding then
        if SilentAimSettings.Triggerbot then
            local t = Mouse.Target
            if t and t.Parent and t:FindFirstAncestorOfClass("Model"):FindFirstChild("Humanoid") and t:FindFirstAncestorOfClass("Model") ~= LocalPlayer.Character then
                mouse1click()
            end
        end
        if SilentAimSettings.Aimbot then
            local target, closest = nil, SilentAimSettings.AimFOV
            for _, v in pairs(Players:GetPlayers()) do
                if v ~= LocalPlayer and v.Character and v.Character:FindFirstChild("Head") then
                    local pos, onScreen = Camera:WorldToViewportPoint(v.Character.Head.Position)
                    if onScreen then
                        local mag = (Vector2.new(pos.X, pos.Y) - mouseLoc).Magnitude
                        if mag < closest then 
                            closest = mag 
                            target = v.Character.Head 
                        end
                    end
                end
            end
            if target then
                local pos = Camera:WorldToViewportPoint(target.Position)
                mousemoverel((pos.X - mouseLoc.X) * SilentAimSettings.AimSmooth, (pos.Y - mouseLoc.Y) * SilentAimSettings.AimSmooth)
            end
        end
    end

    -- 3. ESP Framework Frame Processing
    local showNames = Toggles.ESP_Name.Value
    local showWeapons = Toggles.ESP_Weapon.Value
    local showDist = Toggles.ESP_Distance.Value
    local showBox = Toggles.ESP_Box.Value
    local showSkele = Toggles.ESP_Skeleton.Value
    local showHealth = Toggles.ESP_Healthbar.Value
    local showHighlight = Toggles.ESP_Highlight.Value
    local showTeammates = Toggles.ESP_ShowTeammates.Value
    local betterBounds = Toggles.ESP_BetterBounds.Value
    
    local skeletonThick = Options.ESP_SkeletonThickness.Value
    local textSizeValue = Options.ESP_TextSize.Value
    
    local nameCol1, nameCol2 = Options.NameColor1.Value, Options.NameColor2.Value
    local weapCol1, weapCol2 = Options.WeaponColor1.Value, Options.WeaponColor2.Value
    local distCol1, distCol2 = Options.DistanceColor1.Value, Options.DistanceColor2.Value
    local boxCol1, boxCol2 = Options.BoxColor1.Value, Options.BoxColor2.Value
    local skeleCol1, skeleCol2 = Options.SkeletonColor1.Value, Options.SkeletonColor2.Value
    local hpCol1, hpCol2 = Options.HealthbarColor1.Value, Options.HealthbarColor2.Value
    local hlCol1, hlCol2 = Options.HighlightColor1.Value, Options.HighlightColor2.Value

    local localRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    local localPos = localRoot and localRoot.Position or Camera.Focus.Position

    for player, data in pairs(ESP_Cache) do
        local char = player.Character
        local root = data.CachedParts["HumanoidRootPart"]
        
        if not root or not root.Parent then
            root = char and char:FindFirstChild("HumanoidRootPart")
            if root then data.CachedParts["HumanoidRootPart"] = root end
        end

        local hum = char and char:FindFirstChildOfClass("Humanoid")

        if not (char and root and hum and hum.Health > 0) or (player.Team == LocalPlayer.Team and player.Team ~= nil and not showTeammates) then
            data.BoxOutline.Visible = false data.Box.Visible = false
            data.HealthBarBg.Visible = false data.HealthBar.Visible = false
            data.NameText.Visible = false data.WeaponText.Visible = false data.DistanceText.Visible = false
            data.Highlight.Adornee = nil
            for i = 1, #data.SkeletonLines do data.SkeletonLines[i].Visible = false end
            continue
        end

        local rootPos = root.Position
        local dist = (rootPos - localPos).Magnitude

        local offset = betterBounds and 0.5 or 0
        local topScreen, onScreen1 = Camera:WorldToViewportPoint(rootPos + Vector3.new(0, 3 + offset, 0))
        local bottomScreen, onScreen2 = Camera:WorldToViewportPoint(rootPos + Vector3.new(0, -3.5 - offset, 0))

        if onScreen1 or onScreen2 then
            local boxHeight = math.abs(topScreen.Y - bottomScreen.Y)
            local boxWidth = boxHeight * 0.62
            local boxX = topScreen.X - (boxWidth / 2)
            local boxY = topScreen.Y

            -- Boxes
            if showBox then
                local sz = Vector2.new(boxWidth, boxHeight)
                local pos = Vector2.new(boxX, boxY)
                data.BoxOutline.Size = sz data.BoxOutline.Position = pos data.BoxOutline.Color = boxCol2 data.BoxOutline.Visible = true
                data.Box.Size = sz data.Box.Position = pos data.Box.Color = boxCol1 data.Box.Visible = true
            else
                data.BoxOutline.Visible = false data.Box.Visible = false
            end

            -- Healthbars
            if showHealth then
                local pct = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
                local barX = boxX - 5
                data.HealthBarBg.From = Vector2.new(barX, boxY)
                data.HealthBarBg.To = Vector2.new(barX, boxY + boxHeight)
                data.HealthBarBg.Visible = true
                
                data.HealthBar.From = Vector2.new(barX, boxY + boxHeight)
                data.HealthBar.To = Vector2.new(barX, (boxY + boxHeight) - (boxHeight * pct))
                data.HealthBar.Color = hpCol2:Lerp(hpCol1, pct)
                data.HealthBar.Visible = true
            else
                data.HealthBarBg.Visible = false data.HealthBar.Visible = false
            end

            -- Texts
            local textOffset = 0
            if showNames then
                data.NameText.Text = player.DisplayName or player.Name
                data.NameText.Size = textSizeValue data.NameText.Color = nameCol1 data.NameText.OutlineColor = nameCol2
                data.NameText.Position = Vector2.new(topScreen.X, boxY - textSizeValue - 3)
                data.NameText.Visible = true
            else data.NameText.Visible = false end

            if showWeapons then
                local tool = char:FindFirstChildOfClass("Tool")
                data.WeaponText.Text = tool and tool.Name or "Hands"
                data.WeaponText.Size = textSizeValue data.WeaponText.Color = weapCol1 data.WeaponText.OutlineColor = weapCol2
                data.WeaponText.Position = Vector2.new(topScreen.X, boxY + boxHeight + textOffset)
                textOffset = textOffset + textSizeValue
            else data.WeaponText.Visible = false end

            if showDist then
                data.DistanceText.Text = tostring(math.round(dist)) .. " studs"
                data.DistanceText.Size = textSizeValue data.DistanceText.Color = distCol1 data.DistanceText.OutlineColor = distCol2
                data.DistanceText.Position = Vector2.new(topScreen.X, boxY + boxHeight + textOffset)
                data.DistanceText.Visible = true
            else data.DistanceText.Visible = false end

            -- Skeleton Handling Processing Pipeline
            if showSkele and dist < 175 then
                if next(data.CachedParts) == nil then CachePlayerParts(data, char, data.RigType) end
                
                local boneStructure = (data.RigType == Enum.HumanoidRigType.R15) and R15_Bones or R6_Bones
                local framePositions = {}
                local lineIdx = 1

                local function getPartScreenVec(partName)
                    if framePositions[partName] ~= nil then 
                        return framePositions[partName] 
                    end
                    local partInstance = data.CachedParts[partName]
                    if partInstance and partInstance.Parent then
                        local p, onScreen = Camera:WorldToViewportPoint(partInstance.Position)
                        if onScreen then
                            framePositions[partName] = Vector2.new(p.X, p.Y)
                            return framePositions[partName]
                        end
                    end
                    framePositions[partName] = false
                    return false
                end

                for i = 1, #boneStructure.Spine do
                    local pair = boneStructure.Spine[i]
                    local a, b = getPartScreenVec(pair[1]), getPartScreenVec(pair[2])
                    if a and b then
                        local l = data.SkeletonLines[lineIdx]
                        l.From = a l.To = b l.Color = skeleCol2 l.Thickness = skeletonThick l.Visible = true
                        lineIdx = lineIdx + 1
                    end
                end

                for i = 1, #boneStructure.Limbs do
                    local pair = boneStructure.Limbs[i]
                    local a, b = getPartScreenVec(pair[1]), getPartScreenVec(pair[2])
                    if a and b then
                        local l = data.SkeletonLines[lineIdx]
                        l.From = a l.To = b l.Color = skeleCol1 l.Thickness = skeletonThick l.Visible = true
                        lineIdx = lineIdx + 1
                    end
                end

                for i = lineIdx, #data.SkeletonLines do data.SkeletonLines[i].Visible = false end
            else
                for i = 1, #data.SkeletonLines do data.SkeletonLines[i].Visible = false end
            end

            -- Adornee Highlights Engine
            if showHighlight then
                data.Highlight.Adornee = char data.Highlight.FillColor = hlCol1 data.Highlight.OutlineColor = hlCol2 data.Highlight.FillTransparency = 0.4
            else data.Highlight.Adornee = nil end
        else
            data.BoxOutline.Visible = false data.Box.Visible = false
            data.HealthBarBg.Visible = false data.HealthBar.Visible = false
            data.NameText.Visible = false data.WeaponText.Visible = false data.DistanceText.Visible = false
            data.Highlight.Adornee = nil
            for i = 1, #data.SkeletonLines do data.SkeletonLines[i].Visible = false end
        end
    end
end)
