if not game:IsLoaded() then game.Loaded:Wait() end

local repo = 'https://raw.githubusercontent.com/violin-suzutsuki/LinoriaLib/main/'
local Library = loadstring(game:HttpGet(repo .. 'Library.lua'))()
local ThemeManager = loadstring(game:HttpGet(repo .. 'addons/ThemeManager.lua'))()

local v8 = game:GetService('Players') 
local v10 = game:GetService('RunService') 
local v11 = v8.LocalPlayer 
local v12 = v11.Character or v11.CharacterAdded:Wait() 
local v13 = v12:WaitForChild('HumanoidRootPart') 

-- Logic Variables
local voidConn, orbConn, slingConn, childAddedConn, childRemovedConn
local angle = 0
local targetPos = CFrame.new(9000, 9000, 9000)
local projectiles = {}

-- Utility
local function getClosest()
    local closest, dist = nil, math.huge
    for _, p in pairs(v8:GetPlayers()) do
        if p ~= v11 and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local sameTeam = (v11.Team ~= nil and p.Team ~= nil and v11.Team == p.Team)
            if not sameTeam then
                local d = (v13.Position - p.Character.HumanoidRootPart.Position).Magnitude
                if d < dist then dist = d closest = p end
            end
        end
    end
    return closest
end

-- Window Setup
local Window = Library:CreateWindow({
    Title = 'tomik lua',
    Center = true,
    AutoShow = true,
})

local MainTab = Window:AddTab('Main')

-- LEFT SIDE: FEATURES
local LeftGroupBox = MainTab:AddLeftGroupbox('Combat / Utility')

LeftGroupBox:AddToggle('VoidToggle', {
    Text = 'Void Mode',
    Default = false,
    Callback = function(Value)
        if Value then
            voidConn = v10.Heartbeat:Connect(function()
                if v13 then v13.CFrame = CFrame.new(v13.Position.X + 2e15, 999999, v13.Position.Z + 2e15) end
            end)
        else
            if voidConn then voidConn:Disconnect() voidConn = nil end
        end
    end
})

LeftGroupBox:AddToggle('OrbitToggle', {
    Text = 'Orbit Closest',
    Default = false,
    Callback = function(Value)
        if Value then
            orbConn = v10.Heartbeat:Connect(function()
                local target = getClosest()
                if target and target.Character then
                    local hrp = target.Character:FindFirstChild("HumanoidRootPart")
                    if hrp then
                        angle = angle + 0.18
                        v13.CFrame = CFrame.new(hrp.Position + Vector3.new(math.cos(angle)*6, 4, math.sin(angle)*6), hrp.Position)
                    end
                end
            end)
        else
            if orbConn then orbConn:Disconnect() orbConn = nil end
        end
    end
})

LeftGroupBox:AddToggle('SlingToggle', {
    Text = 'Bypass Sling',
    Default = false,
    Callback = function(Value)
        if Value then
            childAddedConn = workspace.ChildAdded:Connect(function(o)
                if not o:IsA("BasePart") then return end
                if o.Name == "CoreProjectile" or (o.Name == "Part" and o.AssemblyLinearVelocity.Magnitude > 50) then
                    projectiles[o] = true
                end
            end)
            childRemovedConn = workspace.ChildRemoved:Connect(function(o) projectiles[o] = nil end)
            slingConn = v10.Heartbeat:Connect(function()
                pcall(function()
                    for _, p in pairs(v8:GetPlayers()) do
                        if p ~= v11 and p.Character then
                            local h = p.Character:FindFirstChild("HumanoidRootPart")
                            if h then h.CFrame = targetPos h.AssemblyLinearVelocity = Vector3.zero end
                        end 
                    end
                    for p in pairs(projectiles) do
                        if p and p.Parent then p.CFrame = targetPos p.AssemblyLinearVelocity = Vector3.zero
                        else projectiles[p] = nil end 
                    end
                end)
            end)
        else
            if slingConn then slingConn:Disconnect() slingConn = nil end
            if childAddedConn then childAddedConn:Disconnect() childAddedConn = nil end
            if childRemovedConn then childRemovedConn:Disconnect() childRemovedConn = nil end
            projectiles = {}
        end
    end
})

-- RIGHT SIDE: THEME & COLOR
local RightGroupBox = MainTab:AddRightGroupbox('Menu Customization')

ThemeManager:SetLibrary(Library)
ThemeManager:ApplyToGroupbox(RightGroupBox) -- Puts colors in the main tab!

-- Right Control Keybind
game:GetService("UserInputService").InputBegan:Connect(function(input, processed)
    if not processed and input.KeyCode == Enum.KeyCode.RightControl then
        Library:Toggle()
    end
end)

v11.CharacterAdded:Connect(function(c) v12 = c v13 = c:WaitForChild('HumanoidRootPart') end)
Library:Notify('tomik lua loaded! Colors are on the right.')
